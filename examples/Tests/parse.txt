PROC tparse:(f$, rel$)
	LOCAL ret$(32), off%(6)
	ret$ = PARSE$(f$, rel$, off%())
	PRINT "PARSE", f$, rel$, "=", ret$,
	PRINT off%(1), off%(2), off%(3), off%(4), off%(5), off%(6)
ENDP
