PROC expectEqual:(a%, b%, description$)
	PRINT description$;
	PRINT ": ";
	IF a% = b%
		PRINT "Success"
	ELSE
		PRINT "Failure"
		PRINT "  ";
		PRINT a%;
		PRINT " != ";
		PRINT b%
		GET
		STOP
	ENDIF
ENDP
